  loading="lazy" width="100%" height="500" frameborder="0" marginheight="0" marginwidth="0" title="Tally Form">
</iframe>

<script>
async function enrichTallyForm() {
    const ipData = await fetch('https://api.ipify.org?format=json').then(res => res.json());
    const userAgent = navigator.userAgent;

    const frame = document.getElementById('tally-frame');
    let src = frame.getAttribute('data-tally-src');

    src += `&ip_address=${encodeURIComponent(ipData.ip)}&user_agent=${encodeURIComponent(userAgent)}`;
    frame.setAttribute('src', src);
}
enrichTallyForm();
</script>
